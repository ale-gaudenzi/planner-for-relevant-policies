565 template<class T, class tree_node_allocator>

589  {

592  iter ret=it;
593  ret.skip_children();
594  ++ret;
595  erase_children(it);
596  if(cur->prev_sibling==0) {
597  cur->parent->first_child=cur->next_sibling;
598  }
599  else {
600  cur->prev_sibling->next_sibling=cur->next_sibling;
601  }
602  if(cur->next_sibling==0) {
603  cur->parent->last_child=cur->prev_sibling;
604  }
605  else {
606  cur->next_sibling->prev_sibling=cur->prev_sibling;
607  }
608 
609  kp::destructor(&cur->data);
610  alloc_.deallocate(cur,1);
611  return ret;
612  }

655 template <class T, class tree_node_allocator>
656 typename tree<T, tree_node_allocator>::fixed_depth_iterator tree<T, tree_node_allocator>::begin_fixed(const iterator_base& pos, unsigned int dp) const
657  {
658  typename tree<T, tree_node_allocator>::fixed_depth_iterator ret;
659  ret.top_node=pos.node;
660 
661  tree_node *tmp=pos.node;
662  unsigned int curdepth=0;
663  while(curdepth<dp) { // go down one level
664  while(tmp->first_child==0) {
665  if(tmp->next_sibling==0) {
666  // try to walk up and then right again
667  do {
668  if(tmp==ret.top_node)
669  throw std::range_error("tree: begin_fixed out of range");
670  tmp=tmp->parent;
671  if(tmp==0)
672  throw std::range_error("tree: begin_fixed out of range");
673  --curdepth;
674  } while(tmp->next_sibling==0);
675  }
676  tmp=tmp->next_sibling;
677  }
678  tmp=tmp->first_child;
679  ++curdepth;
680  }
681 
682  ret.node=tmp;
683  return ret;
684  }

1486  // are inserted, then this routine should be called 'stable_sort'.
1487  std::multiset<tree_node *, compare_nodes<StrictWeakOrdering> > nodes(comp);
1488  sibling_iterator it=from, it2=to;
1489  while(it != to) {
1490  nodes.insert(it.node);
1491  ++it;
1492  }
1493  // reassemble
1494  --it2;
1495

1497  tree_node *prev=from.node->prev_sibling;
1498  tree_node *next=it2.node->next_sibling;

1503  }
1504  else prev->next_sibling=(*nit);
1505 
1506  --eit;
1507  while(nit!=eit) {
1508  (*nit)->prev_sibling=prev;
1509  if(prev)
1510  prev->next_sibling=(*nit);
1511  prev=(*nit);
1512  ++nit;
1513  }

1515  if(prev)
1516  prev->next_sibling=(*eit);
1517

1519  (*eit)->next_sibling=next;
1520  (*eit)->prev_sibling=prev; // missed in the loop above
1521  if(next==0) {
1522  if((*eit)->parent!=0) // to catch "sorting the head" situations, when there is no parent
1523  (*eit)->parent->last_child=(*eit);
1524  }
1525  else next->prev_sibling=(*eit);
1526 
1527  if(deep) { // sort the children of each node too
1528  sibling_iterator bcs(*nodes.begin());
1529  sibling_iterator ecs(*eit);
1530  ++ecs;
1531  while(bcs!=ecs) {
1532  sort(begin(bcs), end(bcs), comp, deep);
